qb_stats <- pbp_data |>
  filter(
    play_type == "pass",
    !is.na(passer_player_name),
    season_type == "REG"
  ) |>
  group_by(
    player_name = passer_player_name,
    team = posteam 
  ) |>
  summarise(
    attempts = sum(pass_attempt, na.rm = TRUE),
    completions = sum(complete_pass, na.rm = TRUE),
    passing_epa = sum(epa, na.rm = TRUE),
    .groups = 'drop'
  ) |>
  filter(attempts >= 400) |> 
  mutate(
    comp_perc = (completions / attempts) * 100,
    epa_per_db = passing_epa / attempts
  ) |>
  filter(team %in% c("GB", "KC", "BUF", "BAL", "SF", "DET", "JAX", "LAC", "SEA", "HOU", "DAL"))

name_mapping <- c(
    "J.Love" = "Jordan Love",
    "P.Mahomes" = "Patrick Mahomes",
    "J.Allen" = "Josh Allen",
    "L.Jackson" = "Lamar Jackson",
    "C.Stroud" = "C.J. Stroud", 
    "J.Goff" = "Jared Goff",
    "B.Purdy" = "Brock Purdy",
    "T.Lawrence" = "Trevor Lawrence",
    "J.Herbert" = "Justin Herbert",
    "G.Smith" = "Geno Smith",
    "D.Prescott" = "Dak Prescott"
)

qb_stats_standardized <- qb_stats |>
    mutate(
        player_name_std = recode(player_name, !!!name_mapping)
    ) |>
    filter(player_name_std %in% name_mapping) |>
    select(player_name_std, team, attempts, comp_perc, epa_per_db)

min_comp <- min(qb_plot1_data$comp_perc, na.rm = TRUE)
max_comp <- max(qb_plot1_data$comp_perc, na.rm = TRUE)
min_drops <- min(qb_plot1_data$total_team_drops, na.rm = TRUE)
max_drops <- max(qb_plot1_data$total_team_drops, na.rm = TRUE)

qb_plot1_scaled <- qb_plot1_data |>
  mutate(
    scaled_drops = min_comp + (max_comp - min_comp) * (1 - (total_team_drops - min_drops) / (max_drops - min_drops))
  ) |>
  arrange(desc(comp_perc)) |>
  mutate(
    qb_name_fct = factor(player_name_std, levels = player_name_std)
  )

When you look at this upcoming graph you are looking at QB efficiency with EPA (expected points after play) compared with team drop percent. When looking at this we can find how dangerous A QB is while also looking at how a receiver effects them in the passing game. We find that Jordan Love likes to push the ball down the field for big plays while he team is around the middle of the pack with drops.

color_line <- "#00473E" 
color_average <- "#FFC72C" 
color_absence_shade <- "red" 
average_epa <- mean(love_plot_data$weekly_epa_per_db, na.rm = TRUE)

every_other_break <- function(x) {
  x[seq(1, length(x), by = 2)]
}

p2 <- ggplot(love_plot_data, aes(x = game_id, y = weekly_epa_per_db, group = season)) +
  
  geom_area(
    aes(y = ifelse(high_absence_flag, weekly_epa_per_db, NA), fill = "High Absence"), 
    alpha = 0.5
  ) +
  geom_hline(yintercept = average_epa, linetype = "dashed", color = color_average) +
  geom_line(aes(group = season), color = color_line, size = 1.2) +
  geom_point(color = color_line, size = 3) +
  geom_text_repel(
    data = subset(love_plot_data, high_absence_flag == TRUE),
    aes(label = paste0("Wk", week, ": Missed ", key_players_missed)),
    nudge_y = 0.1, size = 3, min.segment.length = 0
  ) +
  
  scale_x_discrete(
    breaks = every_other_break, 
    guide = guide_axis(n.dodge = 1)
  ) +
  
  labs(
    title = "Jordan Love's Weekly Performance (2023-2024)",
    subtitle = paste0("EPA/Dropback. Overall Average: ", round(average_epa, 3)),

    x = "Game Week (2023_Week | 2024_Week)",

So overall by looking at this you can say that Jordan Love has shown flashes of being elite but needs to be more consistent. So is Jordan Love a top 5 QB in the NFL not quite but if he keeps preforming he can end up there, but he is in the top 10.
